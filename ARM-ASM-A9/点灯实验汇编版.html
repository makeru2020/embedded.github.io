点灯实验汇编版


.text


@ 设置LED灯的配置寄存器 GPX2CON 为输出状态
ldr r0,=0x11000c40

ldr r1,[r0]
bic r1,#0xf0000000  
@表示最高的一位中的四个字节全部清零
orr r1,#0x10000000
str r1,[r0]         
@以上操作表示只改变最高的4位（7号位） 而不去影响其他值

@ 设置GPX1CON为输出状态
ldr r0,=0x11000c20
ldr r1,[r0]
bic r1,#0x0000000f  
orr r1,#0x00000001
str r1,[r0]         

@ 设置GPF3CON为输出状态
ldr r0,=0x114001E0
ldr r1,[r0]
bic r1,#0x000f0000 
orr r1,#0x00010000
str r1,[r0] 

@ 设置GPF3CON_5为输出状态
ldr r0,=0x114001E0
ldr r1,[r0]
bic r1,#0x00F00000 
orr r1,#0x00100000
str r1,[r0] 


loop:
@设置LED灯输出高电平GPX2DAT
ldr r0,=0x11000c44
ldr r1,[r0]
orr r1,#0x80
str r1,[r0]

bl delay1s

@设置LED灯输出低电平GPX2DAT
ldr r0,=0x11000c44
ldr r1,[r0]
bic r1,#0x80
str r1,[r0]



@设置LED灯输出高电平GPX1DAT
ldr r0,=0x11000c24
ldr r1,[r0]
orr r1,#0x01
str r1,[r0]

bl delay1s

@设置LED灯输出低电平GPX1DAT
ldr r0,=0x11000c24
ldr r1,[r0]
bic r1,#0x01
str r1,[r0]



@设置LED灯输出高电平GPF3DAT
ldr r0,=0x114001E4
ldr r1,[r0]
orr r1,#0x10
str r1,[r0]

bl delay1s

@设置LED灯输出低电平GPF3DAT
ldr r0,=0x114001E4
ldr r1,[r0]
bic r1,#0x10
str r1,[r0]



@设置LED灯输出高电平GPF3DAT_5
ldr r0,=0x114001E4
ldr r1,[r0]
orr r1,#0x20
str r1,[r0]

bl delay1s

@设置LED灯输出低电平GPX2DAT_5
ldr r0,=0x114001E4
ldr r1,[r0]
bic r1,#0x20
str r1,[r0]

bl delay1s
b loop

delay1s:
ldr r4,=0x1fffffff
delay1s_loop:
sub r4,r4,#1
cmp r4,#0
bne delay1s_loop
mov pc,lr
